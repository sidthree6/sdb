sdb_query

sdb_query ( string $query )
Description:

sdb_query () Perform SQL query to communicate with SDB API . (Click here to learn about SQL Query)

Parameters:

$query | Send SQL Query to the database

Examples for sdb_query

Example 1: Creating Table, Inserting Table and Selecting Table

<?php
require_once("db.php");

sdb_init();

sdb_open_db("database_name");

sdb_query("CREATE TABLE table_name (id PKEY, name VARCHAR(99))");
// Create the table
sdb_query("INSERT INTO table_name (name) VALUES ('first')");
sdb_query("INSERT INTO table_name (name) VALUES ('second')");
sdb_query("INSERT INTO table_name (name) VALUES ('third')");
// Insert into table

$rows = sdb_query("SELECT * FROM table_name");

foreach($rows['name'] as $key=>$value)
{
    echo $rows['id'][$key]." ".$rows['name'][$key]."<br/>";
}

/*
Output
-------------
| 1  first  |
| 2  second |
| 3  third  |
-------------
*/
?>
